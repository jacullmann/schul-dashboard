RR.vue: <template>
          <div v-if="isVisible" class="card-container">
            <Transition name="mac-window">
              <div
                  v-show="isShown"
                  class="draggable-card"
                  ref="cardRef"
                  :style="cardStyle"
                  @mousedown.stop="startDrag"
                  @touchstart.stop="startDrag"
              >
                <div class="card-header">
                  <div class="card-title">Schwebendes Fenster</div>
                  <div class="card-controls">
                    <button
                        @click.stop="toggleCollapse"
                        :disabled="state.isDocked"
                        class="control-btn minimize"
                        title="Einklappen"
                    >
                      –
                    </button>
                    <button
                        @click.stop="startCloseAnimation"
                        class="control-btn close"
                        title="Schließen"
                    >
                      X
                    </button>
                  </div>
                </div>

                <div v-show="!state.isCollapsed" class="card-content">
                  <p>Bewege mich mit der Maus! (Innerhalb des Viewports)</p>
                  <p>Lasse mich nahe eines Bildschirmrands los, um anzudocken.</p>
                </div>
              </div>
            </Transition>
          </div>
        </template>

        <script setup lang="ts">
        import { ref, reactive, computed, onMounted, onUnmounted, watch } from 'vue';

        // --- Props und Emits ---
        const props = defineProps<{
          isVisible: boolean; // Steuert das Öffnen/Schließen von außen
        }>();

        const emit = defineEmits(['close']);

        // --- Konstanten und Konfiguration ---
        const SNAP_THRESHOLD = 20;    // Pixel-Abstand zum Rand, um anzudocken
        const DEFAULT_WIDTH = 400;  // Standardbreite
        const DEFAULT_HEIGHT = 300; // Standardhöhe
        const HEADER_HEIGHT = 40;   // Höhe der Kopfzeile (aus CSS)

        // --- Zustand ---
        const isShown = ref(props.isVisible); // Für die v-show Transition
        const cardRef = ref<HTMLElement | null>(null);

        interface CardState {
          x: number;
          y: number;
          width: number;
          height: number;
          isDragging: boolean;
          isDocked: boolean;
          dockedSide: 'none' | 'left' | 'right' | 'top' | 'bottom';
          isCollapsed: boolean; // NEU: Für Einklapp-Zustand
          lastHeight: number;  // NEU: Speichert Höhe vor dem Einklappen
        }

        const state = reactive<CardState>({
          x: window.innerWidth / 2 - DEFAULT_WIDTH / 2,
          y: window.innerHeight / 2 - DEFAULT_HEIGHT / 2,
          width: DEFAULT_WIDTH,
          height: DEFAULT_HEIGHT,
          isDragging: false,
          isDocked: false,
          dockedSide: 'none',
          isCollapsed: false,
          lastHeight: DEFAULT_HEIGHT,
        });

        // --- Watcher für 'isVisible' Prop ---
        watch(() => props.isVisible, (newVal) => {
          if (newVal) {
            // Wenn sichtbar wird, setze auf Standard zurück und zeige
            unDock(); // Setzt Position, Größe und Status zurück
            isShown.value = true;
          }
          // Das Schließen wird durch startCloseAnimation() gehandhabt
        });

        // --- Computed Style (STARK VEREINFACHT) ---
        const cardStyle = computed(() => {
          return {
            '--card-x': `${state.x}px`,
            '--card-y': `${state.y}px`,
            '--card-width': `${state.width}px`,
            '--card-height': `${state.height}px`,
            'cursor': state.isDragging ? 'grabbing' : 'grab',

            // WICHTIG: Immer 'transform' für die Positionierung verwenden.
            // 'top' und 'left' bleiben '0' (siehe CSS).
            'transform': `translate(var(--card-x), var(--card-y))`,

            // WICHTIG: Transition nur anwenden, wenn NICHT gezogen wird.
            // Dies animiert das Andocken, Öffnen, Schließen und Einklappen.
            'transition': state.isDragging ? 'none' : 'all 0.3s ease-in-out',
          };
        });

        // --- Drag-Handler ---
        let dragStart = { x: 0, y: 0 };
        let cardStart = { x: 0, y: 0 };

        function startDrag(event: MouseEvent | TouchEvent) {
          // Nur ziehen, wenn auf den Header geklickt wird
          const target = event.target as HTMLElement;
          const isHeader = target.closest('.card-header');
          if (!isHeader && !('touches' in event)) return;

          const clientX = 'touches' in event ? event.touches[0].clientX : event.clientX;
          const clientY = 'touches' in event ? event.touches[0].clientY : event.clientY;

          // WICHTIG: Wenn angedockt, sofort (ohne Animation) in den
          // "nicht angedockten" Zustand wechseln, zentriert unter dem Cursor.
          if (state.isDocked) {
            // Übergib die Klick-Koordinaten für die Neupositionierung
            unDock(clientX, clientY);
          }

          // Das Ziehen *beginnt* erst, nachdem der Zustand (ggf.) geändert wurde
          state.isDragging = true;
          dragStart = { x: clientX, y: clientY };
          // state.x/y sind jetzt die neuen (ggf. von unDock gesetzten) Koordinaten
          cardStart = { x: state.x, y: state.y };

          event.preventDefault();

          document.addEventListener('mousemove', drag);
          document.addEventListener('mouseup', stopDrag);
          document.addEventListener('touchmove', drag);
          document.addEventListener('touchend', stopDrag);
        }

        function drag(event: MouseEvent | TouchEvent) {
          if (!state.isDragging) return;

          const clientX = 'touches' in event ? event.touches[0].clientX : event.clientX;
          const clientY = 'touches' in event ? event.touches[0].clientY : event.clientY;

          const deltaX = clientX - dragStart.x;
          const deltaY = clientY - dragStart.y;

          let newX = cardStart.x + deltaX;
          let newY = cardStart.y + deltaY;

          // Viewport-Beschränkung (verhindert, dass die Karte aus dem Bild geschoben wird)
          // Beim Ziehen verwenden wir die aktuelle Breite/Höhe
          const cardW = state.width;
          const cardH = state.height;
          const windowW = window.innerWidth;
          const windowH = window.innerHeight;

          state.x = Math.max(0, Math.min(newX, windowW - cardW));
          state.y = Math.max(0, Math.min(newY, windowH - cardH));
        }

        function stopDrag() {
          state.isDragging = false;

          // Prüfe auf Andocken (dies wird jetzt sanft animiert,
          // da 'transition' wieder aktiv ist)
          if (!state.isCollapsed) { // Nur andocken, wenn nicht eingeklappt
            checkAndDock();
          }

          document.removeEventListener('mousemove', drag);
          document.removeEventListener('mouseup', stopDrag);
          document.removeEventListener('touchmove', drag);
          document.removeEventListener('touchend', stopDrag);
        }

        // --- Andock-Logik ---

        function checkAndDock() {
          const x = state.x;
          const y = state.y;
          const windowW = window.innerWidth;
          const windowH = window.innerHeight;
          const cardW = state.width;
          const cardH = state.height;

          // 1. Linker Rand
          if (x < SNAP_THRESHOLD) {
            dock('left');
          }
          // 2. Rechter Rand
          else if (x + cardW > windowW - SNAP_THRESHOLD) {
            dock('right');
          }
          // 3. Oberer Rand
          else if (y < SNAP_THRESHOLD) {
            dock('top');
          }
          // 4. Unterer Rand
          else if (y + cardH > windowH - SNAP_THRESHOLD) {
            dock('bottom');
          }
          // 5. Kein Andocken
          else {
            state.isDocked = false;
            state.dockedSide = 'none';
          }
        }

        function dock(side: 'left' | 'right' | 'top' | 'bottom') {
          state.isDocked = true;
          state.isCollapsed = false; // Andocken erzwingt das Ausklappen
          state.dockedSide = side;

          switch (side) {
            case 'left':
              state.x = 0;
              state.y = 0;
              state.width = window.innerWidth / 2;
              state.height = window.innerHeight;
              break;
            case 'right':
              state.x = window.innerWidth / 2;
              state.y = 0;
              state.width = window.innerWidth / 2;
              state.height = window.innerHeight;
              break;
            case 'top':
              state.x = 0;
              state.y = 0;
              state.width = window.innerWidth;
              state.height = window.innerHeight / 2;
              break;
            case 'bottom':
              state.x = 0;
              state.y = window.innerHeight / 2;
              state.width = window.innerWidth;
              state.height = window.innerHeight / 2;
              break;
          }
        }

        /**
         * Setzt die Karte in den Standardzustand (nicht angedockt) zurück.
         * @param clickX - Optionale X-Koordinate des Klicks (zum Neupositionieren beim Ziehen).
         * @param clickY - Optionale Y-Koordinate des Klicks.
         */
        function unDock(clickX?: number, clickY?: number) {
          state.isDocked = false;
          state.dockedSide = 'none';
          state.isCollapsed = false;

          const oldWidth = state.width;
          const oldHeight = state.height;
          const newWidth = DEFAULT_WIDTH;
          const newHeight = DEFAULT_HEIGHT;

          state.width = newWidth;
          state.height = newHeight;
          state.lastHeight = newHeight;

          if (clickX !== undefined && clickY !== undefined) {
            // --- SCHLÜSSEL-LOGIK GEGEN RUCKELN ---
            // Positioniere die "neue" (kleinere) Karte so, dass der Mauszeiger
            // prozentual an derselben Stelle auf der Karte bleibt.
            const relativeX = clickX - state.x;
            const relativeY = clickY - state.y;
            const percentX = relativeX / oldWidth;
            const percentY = relativeY / oldHeight;

            state.x = clickX - (newWidth * percentX);
            state.y = clickY - (newHeight * percentY);

          } else {
            // Standard-Reset (z.B. beim Öffnen): Zentrieren
            state.x = window.innerWidth / 2 - newWidth / 2;
            state.y = window.innerHeight / 2 - newHeight / 2;
          }

          // Sicherstellen, dass die neue Position im Viewport ist
          state.x = Math.max(0, Math.min(state.x, window.innerWidth - newWidth));
          state.y = Math.max(0, Math.min(state.y, window.innerHeight - newHeight));
        }

        // --- NEU: Einklapp-Logik ---
        function toggleCollapse() {
          // Einklappen nur erlauben, wenn nicht angedockt
          if (state.isDocked) return;

          state.isCollapsed = !state.isCollapsed;
          if (state.isCollapsed) {
            // Einklappen: Höhe speichern und auf Header-Höhe setzen
            state.lastHeight = state.height;
            state.height = HEADER_HEIGHT;
          } else {
            // Ausklappen: Gespeicherte Höhe wiederherstellen
            state.height = state.lastHeight;
          }
        }

        // --- Resize Handler ---
        function handleResize() {
          const windowW = window.innerWidth;
          const windowH = window.innerHeight;

          if (state.isDocked) {
            // Angedockte Position/Größe neu berechnen
            dock(state.dockedSide);
          } else {
            // Sicherstellen, dass die Karte im Viewport bleibt
            state.x = Math.max(0, Math.min(state.x, windowW - state.width));
            state.y = Math.max(0, Math.min(state.y, windowH - state.height));

            // Höhe beibehalten, wenn eingeklappt
            if(state.isCollapsed) {
              state.height = HEADER_HEIGHT;
            }
          }
        }

        // --- Schließ-Animation ---
        function startCloseAnimation() {
          isShown.value = false; // Löst .mac-window-leave-to aus
          setTimeout(() => {
            emit('close'); // Entfernt das Element via v-if
          }, 300); // Muss zur CSS-Transition-Dauer passen
        }

        // --- Lifecycle Hooks ---
        onMounted(() => {
          window.addEventListener('resize', handleResize);
        });

        onUnmounted(() => {
          window.removeEventListener('resize', handleResize);
          // Sicherheitshalber alle globalen Listener entfernen
          document.removeEventListener('mousemove', drag);
          document.removeEventListener('mouseup', stopDrag);
          document.removeEventListener('touchmove', drag);
          document.removeEventListener('touchend', stopDrag);
        });
        </script>

        <style scoped>
        /* Container, der die Positionierung bei v-if/v-show unterstützt */
        .card-container {
          position: absolute;
          top: 0;
          left: 0;

          height: 0;
        }

        .draggable-card {
          position: fixed;
          /* WICHTIG: top/left sind 0. Die Positionierung
             erfolgt AUSSCHLIESSLICH über 'transform'. */
          top: 0;
          left: 0;

          /* Größe wird über CSS-Variablen gesteuert */
          width: var(--card-width);
          height: var(--card-height);

          background-color: white;
          border: 1px solid #ccc;
          border-radius: 10px;
          box-shadow: 0 15px 40px rgba(0, 0, 0, 0.3);
          z-index: 1000000;

          display: flex;
          flex-direction: column;
          overflow: hidden; /* Wichtig für die Einklapp-Animation */
          color: black;

          /* Performance-Optimierung für Animationen */
          will-change: transform, width, height, opacity;
        }

        /* ------------------- MAC-STIL ANIMATION (KORRIGIERT) ------------------- */

        .mac-window-enter-active,
        .mac-window-leave-active {
          /* Die Transition für 'all' (Größe, Position) wird durch :style="cardStyle" gesteuert.
             Hier definieren wir *nur* die Transitions für Opacity und Scale,
             die von der <Transition>-Komponente gesteuert werden. */
          transition: opacity 0.3s ease-in-out, transform 0.3s ease-in-out;
          transform-origin: center; /* Von der Mitte aus skalieren */
        }

        /* * KORREKTUR: Die 'from' und 'to' Zustände MÜSSEN die
         * finale 'translate'-Position beinhalten, damit die Skalierung
         * an der korrekten Endposition stattfindet.
         */

        /* Animation beim Erscheinen (v-enter-from) */
        .mac-window-enter-from {
          opacity: 0;
          transform: translate(var(--card-x), var(--card-y)) scale(0.7);
        }
        /* Animation beim Verschwinden (v-leave-to) */
        .mac-window-leave-to {
          opacity: 0;
          transform: translate(var(--card-x), var(--card-y)) scale(0.7);
          z-index: 1001;
        }

        /* ------------------- HEADER UND STEUERUNG ------------------- */
        .card-header {
          height: 40px; /* Muss zu HEADER_HEIGHT in <script> passen */

          user-select: none;

          border-bottom: 1px solid #ddd;
        }

        .card-title {

        }

Playg.vue: <script setup>
           import { ref } from 'vue';
           import DraggableCard from './RR.vue';

           const showCard = ref(false);

           function openCard() {
             showCard.value = true;
           }

           function handleClose() {
             showCard.value = false;
           }
           </script>

           <template>


               <button @click="openCard" :disabled="showCard" style="padding: 10px 20px; font-size: 16px;">
                 Fenster öffnen
               </button>

               <DraggableCard
                   :is-visible="showCard"
                   @close="handleClose"
               />

           </template>
