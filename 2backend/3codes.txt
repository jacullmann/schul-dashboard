hier die Stundenplan.vue: <script setup lang="ts">
                          import { computed, onMounted, onBeforeUnmount, ref } from "vue";
                          import data from "../schedule.json";

                          type DayKey = string;

                          interface TimeSlot {
                            id: number;
                            label: string;
                            start?: string;
                            end?: string;
                          }
                          interface Item {
                            subject: string;
                            teacher?: string;
                            room?: string;
                            type?: string;
                            className?: string;
                            color?: string;
                            info?: string;
                          }
                          interface Group {
                            label?: string;
                            items: Item[];
                            className?: string;
                          }
                          interface Block {
                            startSlot: number;
                            endSlot: number;
                            items?: Item[];
                            groups?: Group[];
                            className?: string;
                            blockInfo?: string;
                          }
                          interface Dictionary {
                            subjects?: Record<string, { name?: string; className?: string; color?: string }>;
                            tags?: Record<string, { label?: string; className?: string }>;
                          }
                          interface ScheduleJson {
                            meta: {
                              title: string;
                              subtitle?: string;
                              classLabel?: string;
                              locale?: string;
                              highlightToday?: boolean;
                              showEmptySlots?: boolean;
                              showTimeColumn?: boolean;
                              timeColumnLabel?: string;
                              mode?: "compact" | "expanded";
                              responsive?: {
                                mobileBreakpoint?: number;
                                stickyHeader?: boolean;
                                stickyTimeColumn?: boolean;
                              };
                            };
                            ui: {
                              dayLabels?: Record<DayKey, string>;
                              accentBorderWidth?: number;
                              accentBorderStyle?: string;
                              cardRadius?: number;
                              gridGap?: number;
                              groupColumnsDesktop?: number;
                              groupColumnsMobile?: number;
                              hover?: { scale?: number; shadow?: string };
                            };
                            layout: {
                              days: { key: DayKey }[];
                              timeSlots: TimeSlot[];
                            };
                            dictionary?: Dictionary;
                            schedule: Record<DayKey, Block[]>;
                            breaks?: { slotId: number; label?: string; showInTimeColumn?: boolean }[];
                          }

                          const schedule = data as ScheduleJson;

                          // Responsive
                          const width = ref<number>(typeof window !== "undefined" ? window.innerWidth : 1024);
                          const mobileBreakpoint = schedule.meta.responsive?.mobileBreakpoint ?? 768;
                          const isMobile = computed(() => width.value < mobileBreakpoint);
                          function onResize() { width.value = window.innerWidth; }
                          onMounted(() => window.addEventListener("resize", onResize));
                          onBeforeUnmount(() => window.removeEventListener("resize", onResize));

                          // Sichtbarkeit (alle Tage/Slots, keine A/B/Datum-Checks)
                          const visibleDays = computed(() => schedule.layout.days.map(d => d.key));
                          const visibleSlots = computed(() => schedule.layout.timeSlots);

                          // Heute-Highlight
                          const todayKey = computed(() => {
                            const d = new Date();
                            const idx = d.getDay(); // 0 So ... 6 Sa
                            const map = ["So", "Mo", "Di", "Mi", "Do", "Fr", "Sa"];
                            const key = map[idx];
                            return visibleDays.value.includes(key) ? key : null;
                          });

                          // Rowspan Maps
                          const startBlockMap = computed(() => {
                            const map: Record<DayKey, Record<number, Block>> = {};
                            for (const day of visibleDays.value) {
                              map[day] = {};
                              const blocks = schedule.schedule[day] || [];
                              for (const b of blocks) {
                                map[day][b.startSlot] = b;
                              }
                            }
                            return map;
                          });
                          const coveredMap = computed(() => {
                            const map: Record<DayKey, Record<number, boolean>> = {};
                            for (const day of visibleDays.value) {
                              map[day] = {};
                              const blocks = schedule.schedule[day] || [];
                              for (const b of blocks) {
                                for (let s = b.startSlot + 1; s <= b.endSlot; s++) {
                                  map[day][s] = true;
                                }
                              }
                            }
                            return map;
                          });

                          // Break-Lookup
                          const breakBySlot = computed(() => {
                            const byId: Record<number, { slotId: number; label?: string; showInTimeColumn?: boolean }> = {};
                            (schedule.breaks || []).forEach(b => (byId[b.slotId] = b));
                            return byId;
                          });

                          // Dictionary helpers
                          function subjectName(code: string) {
                            const dict = schedule.dictionary?.subjects;
                            const clean = code.replace(/^\*/, "");
                            return dict?.[clean]?.name || clean;
                          }
                          function subjectClass(code: string) {
                            const dict = schedule.dictionary?.subjects;
                            const clean = code.replace(/^\*/, "");
                            return dict?.[clean]?.className || null;
                          }
                          function subjectColor(code: string, override?: string) {
                            if (override) return override;
                            const dict = schedule.dictionary?.subjects;
                            const clean = code.replace(/^\*/, "");
                            return dict?.[clean]?.color || null;
                          }
                          function tagInfo(type?: string) {
                            if (!type) return null;
                            return schedule.dictionary?.tags?.[type] || null;
                          }

                          // Helpers
                          function getBlock(dayKey: DayKey, slotId: number): Block | null {
                            return startBlockMap.value[dayKey]?.[slotId] || null;
                          }
                          </script>

                          <template>
                            <div class="card timetable-wrapper">
                              <div class="header">
                                <div class="title">
                                  <h2>{{ schedule.meta.title }}</h2>
                                  <div class="small">
                                    {{ schedule.meta.subtitle }} · {{ schedule.meta.classLabel }}
                                  </div>
                                </div>
                              </div>

                              <!-- Desktop/Table View -->
                              <div class="table-scroll" v-show="!isMobile">
                                <table class="timetable" :class="[schedule.meta.mode === 'compact' ? 'compact' : 'expanded']">
                                  <thead :class="{ sticky: schedule.meta.responsive?.stickyHeader }">
                                  <tr>
                                    <th v-if="schedule.meta.showTimeColumn" class="timehead" :class="{ sticky: schedule.meta.responsive?.stickyTimeColumn }">
                                      {{ schedule.meta.timeColumnLabel }}
                                    </th>
                                    <th v-for="day in visibleDays" :key="day"
                                        :class="['day-header', { today: schedule.meta.highlightToday && todayKey === day }]">
                                      {{ schedule.ui.dayLabels?.[day] ?? day }}
                                    </th>
                                  </tr>
                                  </thead>

                                  <tbody>
                                  <tr v-for="slot in visibleSlots" :key="slot.id">
                                    <td v-if="schedule.meta.showTimeColumn" class="time" :class="{ sticky: schedule.meta.responsive?.stickyTimeColumn }">
                                      <div class="time-label">
                                        {{ slot.label }}
                                        <span v-if="breakBySlot[slot.id]?.showInTimeColumn" class="break-badge">{{ breakBySlot[slot.id]?.label }}</span>
                                      </div>
                                    </td>

                                    <template v-for="day in visibleDays" :key="day">
                                      <td v-if="!coveredMap[day]?.[slot.id]"
                                          :rowspan="startBlockMap[day]?.[slot.id]?.endSlot
                                                      ? startBlockMap[day][slot.id].endSlot - startBlockMap[day][slot.id].startSlot + 1
                                                      : 1"
                                          class="cell"
                                          :class="[
                                              startBlockMap[day]?.[slot.id]?.className,
                                              { today: schedule.meta.highlightToday && todayKey === day },
                                              { break: !!breakBySlot[slot.id] }
                                            ]"
                                      >
                                        <!-- Pause-Zelle -->
                                        <div v-if="breakBySlot[slot.id]" class="pause">
                                          {{ breakBySlot[slot.id].label || 'Pause' }}
                                        </div>

                                        <!-- Block-Inhalte -->
                                        <div v-else-if="startBlockMap[day]?.[slot.id]" class="block"
                                             :style="{ borderRadius: (schedule.ui.cardRadius || 10) + 'px' }"
                                        >
                                          <div v-if="startBlockMap[day][slot.id].blockInfo" class="info-pin" aria-label="Block-Info">
                                            ⓘ
                                            <div class="info-bubble">{{ startBlockMap[day][slot.id].blockInfo }}</div>
                                          </div>

                                          <!-- Gruppen-Grid -->
                                          <div v-if="startBlockMap[day][slot.id].groups"
                                               class="groups"
                                               :style="{
                                                   gridTemplateColumns: 'repeat(' + (schedule.ui.groupColumnsDesktop || 2) + ', 1fr)',
                                                   gap: (schedule.ui.gridGap || 8) + 'px'
                                                 }"
                                          >
                                            <div v-for="(group, gi) in startBlockMap[day][slot.id].groups" :key="gi" class="group">
                                              <div v-if="group.label" class="group-label small">{{ group.label }}</div>
                                              <div v-for="(item, ii) in group.items" :key="ii"
                                                   class="lesson"
                                                   :class="[ item.className, subjectClass(item.subject) ]"
                                                   :style="{
                                                       borderLeftColor: subjectColor(item.subject, item.color) || 'transparent',
                                                       borderLeftWidth: (schedule.ui.accentBorderWidth || 4) + 'px',
                                                       borderLeftStyle: schedule.ui.accentBorderStyle || 'solid'
                                                     }"
                                              >
                                                <div class="lesson-top">
                                                  <strong class="subject">{{ subjectName(item.subject) }}</strong>
                                                  <span v-if="tagInfo(item.type)" class="tag" :class="tagInfo(item.type)?.className">
                                                      {{ tagInfo(item.type)?.label }}
                                                    </span>
                                                </div>
                                                <div class="lesson-meta small">
                                                  <span v-if="item.teacher" class="teacher">{{ item.teacher }}</span>
                                                  <span v-if="item.teacher && item.room"> · </span>
                                                  <span v-if="item.room" class="room">{{ item.room }}</span>
                                                </div>
                                                <div v-if="item.info" class="info-pin" aria-label="Info">
                                                  ⓘ
                                                  <div class="info-bubble">{{ item.info }}</div>
                                                </div>
                                              </div>
                                            </div>
                                          </div>

                                          <!-- Einzel-Items -->
                                          <div v-else class="items" :style="{ gap: (schedule.ui.gridGap || 8) + 'px' }">
                                            <div v-for="(item, ii) in startBlockMap[day][slot.id].items" :key="ii"
                                                 class="lesson"
                                                 :class="[ item.className, subjectClass(item.subject) ]"
                                                 :style="{
                                                     borderLeftColor: subjectColor(item.subject, item.color) || 'transparent',
                                                     borderLeftWidth: (schedule.ui.accentBorderWidth || 4) + 'px',
                                                     borderLeftStyle: schedule.ui.accentBorderStyle || 'solid'
                                                   }"
                                            >
                                              <div class="lesson-top">
                                                <strong class="subject">{{ subjectName(item.subject) }}</strong>
                                                <span v-if="tagInfo(item.type)" class="tag" :class="tagInfo(item.type)?.className">
                                                    {{ tagInfo(item.type)?.label }}
                                                  </span>
                                              </div>
                                              <div class="lesson-meta small">
                                                <span v-if="item.teacher" class="teacher">{{ item.teacher }}</span>
                                                <span v-if="item.teacher && item.room"> · </span>
                                                <span v-if="item.room" class="room">{{ item.room }}</span>
                                              </div>
                                              <div v-if="item.info" class="info-pin" aria-label="Info">
                                                ⓘ
                                                <div class="info-bubble">{{ item.info }}</div>
                                              </div>
                                            </div>
                                          </div>
                                        </div>

                                        <!-- Leere Slots -->
                                        <div v-else-if="schedule.meta.showEmptySlots" class="empty small">—</div>
                                      </td>
                                    </template>
                                  </tr>
                                  </tbody>
                                </table>
                              </div>

                              <!-- Mobile/Card View -->
                              <div class="mobile-list" v-show="isMobile">
                                <div v-for="day in visibleDays" :key="day" class="mobile-day">
                                  <div class="mobile-day-header" :class="{ today: schedule.meta.highlightToday && todayKey === day }">
                                    {{ schedule.ui.dayLabels?.[day] ?? day }}
                                  </div>

                                  <div class="mobile-slots">
                                    <div v-for="slot in visibleSlots" :key="slot.id" class="mobile-slot">
                                      <div class="mobile-time small">
                                        {{ slot.label }}
                                        <span v-if="breakBySlot[slot.id]?.showInTimeColumn" class="break-badge">{{ breakBySlot[slot.id]?.label }}</span>
                                      </div>

                                      <div v-if="breakBySlot[slot.id]" class="pause-card">
                                        {{ breakBySlot[slot.id].label || 'Pause' }}
                                      </div>

                                      <div v-else-if="getBlock(day, slot.id)" class="mobile-block">

                                                   borderLeftColor: subjectColor(item.subject, item.color) || 'transparent',
