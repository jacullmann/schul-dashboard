         import { ref, computed } from 'vue';

         const STORAGE_KEY = 'app_auth_token';
         const STORAGE_EXPIRES_KEY = 'app_auth_expires';

         const token = ref<string | null>(null);

         function now() { return Date.now(); }
         function inThirtyDaysMs() { return 30 * 24 * 60 * 60 * 1000; }

         function loadFromStorage() {
             const t = localStorage.getItem(STORAGE_KEY);
             const e = localStorage.getItem(STORAGE_EXPIRES_KEY);
             if (!t || !e) {
                 token.value = null;
                 clearStorage();
                 return;
             }
             const expires = Number(e);
             if (Number.isNaN(expires) || now() > expires) {
                 token.value = null;
                 clearStorage();
                 return;
             }
             token.value = t;
         }

         function clearStorage() {
             localStorage.removeItem(STORAGE_KEY);
             localStorage.removeItem(STORAGE_EXPIRES_KEY);
         }

         loadFromStorage();

         export function useAuth() {
             const isAuthenticated = computed(() => !!token.value);

             async function loginWithCode(code: string) {
                 if (!code) return { ok: false, error: 'Bitte gib einen Code ein' };

                 try {
                     const resp = await fetch('/api/auth/check-code', {
                         method: 'POST',
                         headers: { 'Content-Type': 'application/json' },
                         body: JSON.stringify({ code })
                     });

                     const data = await resp.json().catch(() => ({}));

                     if (!resp.ok || !data?.ok) {
                         return { ok: false, error: data?.error || 'Ungültiger Code' };
                     }

                     // store token and expiry locally (same behaviour as before)
                     const t = data.token;
                     const expires = now() + inThirtyDaysMs();
                     token.value = t;
                     localStorage.setItem(STORAGE_KEY, t);
                     localStorage.setItem(STORAGE_EXPIRES_KEY, String(expires));
                     window.dispatchEvent(new Event('auth-changed'));

                     return { ok: true };
                 } catch (err) {
                     return { ok: false, error: 'Netzwerkfehler' };
                 }
             }

             function logout() {
                 token.value = null;
                 clearStorage();
                 window.dispatchEvent(new Event('auth-changed'));
             }

             function refreshExpiry() {
                 if (!token.value) return;
                 const expires = now() + inThirtyDaysMs();
                 localStorage.setItem(STORAGE_EXPIRES_KEY, String(expires));
             }

             return {
                 token,
                 isAuthenticated,
                 loginWithCode,
                 logout,
                 refreshExpiry,
                 loadFromStorage
             };
         }

           function requireAuth(req, res, next) {

               const now = dayjs();
               const due = dayjs(dueDate);
               const diffDays = due.diff(now, 'day', true);
               if (diffDays < 0) return 'expired';
               if (diffDays < 1) return 'danger';
               if (diffDays < 2) return 'warn';
               if (diffDays < 3) return 'normal';

           // --- simple code-check endpoint with IP lock and brute-force protection ---
           // memory structures (process memory; restarts clear them)
           const failedByIp = new Map(); // ip => { fails: number, lockedUntil: Date | null }

           // config
           const FRONTEND_PROTECTED_CODE = process.env.FRONTEND_PROTECTED_CODE || 'mynewjamaicanlawyer';
           const MAX_FAILED_ATTEMPTS = 5;           // lock after this many failed tries
           const LOCK_DURATION_MS = 30 * 60 * 1000; // 30 minutes lock
           const BURST_RATE_LIMIT_WINDOW_MS = 1000; // 1 second window
           const BURST_MAX_REQUESTS = 5;            // >5 requests/sec from same IP triggers rate limiter

           // per-ip short-term rate limiter (burst protection)
           const burstLimiters = new Map(); // ip => { count, windowStart }

           function checkBurst(ip) {
               const now = Date.now();
               const rec = burstLimiters.get(ip) || { count: 0, windowStart: now };
               if (now - rec.windowStart > BURST_RATE_LIMIT_WINDOW_MS) {
                   // reset window
                   rec.count = 1;
                   rec.windowStart = now;
                   burstLimiters.set(ip, rec);
                   return true;
               }
               rec.count++;
               burstLimiters.set(ip, rec);
               return rec.count <= BURST_MAX_REQUESTS;
           }

           app.post('/api/auth/check-code', body('code').isString().isLength({ min: 1 }), async (req, res) => {
               try {
                   const ip = req.ip || req.connection?.remoteAddress || 'unknown';
                   // burst protection
                   if (!checkBurst(ip)) {
                       return res.status(429).json({ error: 'Too many requests' });
                   }

                   // check lock
                   const info = failedByIp.get(ip) || { fails: 0, lockedUntil: null };
                   if (info.lockedUntil && info.lockedUntil > Date.now()) {
                       return res.status(423).json({ error: 'Temporarily locked due to repeated failures' });
                   } else if (info.lockedUntil && info.lockedUntil <= Date.now()) {
                       // unlock
                       info.fails = 0;
                       info.lockedUntil = null;
                       failedByIp.set(ip, info);
                   }

                   const code = String(req.body.code || '').trim();
                   if (!code) return res.status(400).json({ ok: false, error: 'Code required' });

                   if (code === FRONTEND_PROTECTED_CODE) {
                       // success -> create a signed token for the frontend (shorter or same expiry as you want)
                       const payload = { sub: 'frontend-check', purpose: 'simple_access' };
                       const token = jwt.sign(payload, JWT_SECRET, { expiresIn: '30d' });

                       // reset fail counters for this ip
                       failedByIp.delete(ip);

                       return res.json({ ok: true, token });
                   }

                   // failure: increment fail counter and possibly lock
                   info.fails = (info.fails || 0) + 1;
                   if (info.fails >= MAX_FAILED_ATTEMPTS) {
                       info.lockedUntil = Date.now() + LOCK_DURATION_MS;
                   }
                   failedByIp.set(ip, info);

                   // generic message (avoid revealing whether code exists)
                   return res.status(401).json({ ok: false, error: 'Ungültiger Code' });
               } catch (err) {
                   console.error('POST /api/auth/check-code error', err);
                   return res.status(500).json({ error: 'Server error' });
               }
           });
